00015     //these are because Arduino float doesnt have enough precision, so have to parse twice to get end digits
00016     //ParseGPGLL(lat_deg_nmea_string, &latitudeDeg, &latitudeMin);
00017     //convert the ddmm.mmmm string into dd, mm.mmmm (because there isnt enough precision in an arduino float, we split it into 2)
00018 
00019 //this will only work if strings are terminated by a recognizable null token by strtok, as described here: http://www.acm.uiuc.edu/webmonkeys/book/c_guide/2.14.html
00020 //the idea is to copy all the data beyond the first 3 characters over to a new string, before converting to a float
00021 // Then convert these saved numbers as higher precision points for the float
00022     float smallFraction; //the smaller fractional part of the string, in float
00023     char smallFractionString[10]; //the smaller fractional part of the string
00024     float lowPrecisionGPS; //result of using atof on a high-precision GPS variable (drops the end precision)
00025     double intMinute; //the integer portion of minutes (ie mm of mm.mmmm)
00026     float fractionalMinute; // the fractional portion of minutes (ie 0.mmmm of mm.mmmm)
00027     int i; //counter
00028     double temp; //garbage
00029 
00030     //find degrees and low precision minutes
00031     lowPrecisionGPS = atof(GPGLL_string);
00032     //get the degrees part, and the integer minutes part
00033     // this wont work if somehow there are more than 2 digits of minutes (ie ddmmm.mmmm wll give ddm and mm.mmmm); this shouldnt happen
00034     // this should work for 3 digit degrees though (ie dddmm.mmmm -> ddd and mm.mmmm)
00035     *minute = 100*modf(lowPrecisionGPS / 100.0, degree); //processes ddmm.mm into dd (latDegrees) and mm.mm (latMinutes), low precision minute
00036     //degree stores the integer, minute the fractional part (with low accuracy, ie mm.mm)
00037     //Serial.println(*minute);
00038     //now find high precision minutes
00039 
00040     //find the precise decimal portion (ie the 0.mmmm part), store in fractionalMinute
00041     //drop a few high digits (ddmm.mmmm -> m.mmmm) to make it fit in the precision of a float
00042     i = 0;
00043     while (GPGLL_string[i+4]) {
00044         smallFractionString[i] = GPGLL_string[i+4];
00045         i++;
00046     }
00047     smallFractionString[i]='\0';//append end of string character
00048 
00049     //now there should be 4 decimal points, and 2 integers; this should fit in a float
00050     smallFraction = atof(smallFractionString);
00051     //Serial.println(smallFraction);
00052 
00053     //drop the integer part (m.mmmm -> m and 0.mmmm), save the high precision fraction
00054     fractionalMinute = modf(smallFraction, &temp); //0.mmmm
00055     //Serial.println(fractionalMinute);
00056 
00057     //drop the fraction from the low precision minute variable, save the integer part
00058     temp = modf(*minute, &intMinute);//drop the decimal part, we already have it; save the integer part into intMinute
00059 
00060     //combine the fraction and integer parts to get a high precision minute variable
00061     *minute = intMinute + fractionalMinute;//combine the decimal and integer halfs of the minutes into one number
00062 }
